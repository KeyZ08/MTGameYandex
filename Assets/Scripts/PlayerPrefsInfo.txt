PlayerPrefs

"ActiveLevel" - int текущего (последнего посещенного) уровня
"LastLevel" - int самого дальнего из посещенных уровней
"PlayerMoney" - int количество заработанных монет

"LastInteractiveTraining" - int самого дальнего из пройденных интерактивных обучений
"LastTraining" - int самого дальнего из посещенных обучающих префабов

"FirstGame" - int означающий впервые ли открыта карта (нужен для катсцены)
"EndGame" - int означающий что игра пройдена

"Music" - float значение громкости музыки
"Sounds" - float значение громкости звуков

"InfinityGameSettings" -  json Настройки для бесконечной игры

"Statistic" - str json отображающий историю по дням о колличествах ошибок и решений, а также их подробности
"Shop" - str json хранящий dictionary объектов <имя, (isbuy, isinstall)>
